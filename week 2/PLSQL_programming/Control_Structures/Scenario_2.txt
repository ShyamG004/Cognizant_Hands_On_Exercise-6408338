CREATE TABLE customers1 (
    customer_id NUMBER PRIMARY KEY,
    name VARCHAR2(100),
    balance NUMBER(10,2),
    isVIP CHAR(1) DEFAULT 'N'  -- 'Y' for TRUE, 'N' for FALSE
);
